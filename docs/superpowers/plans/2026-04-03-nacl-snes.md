# NaCl SNES Emulator — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Compile snes9x as a native ARM binary via NaCl to run SNES games at ~85% native speed on Samsung TV 2020 (Tizen 5.5).

**Architecture:** Port snes4nacl (snes9x 1.53 + PPAPI) to Samsung Pepper 63 SDK, cross-compile for ARMv7 inside Docker, embed the .nexe in the existing Tizen web app (.wgt). JS handles menu/ROM selection/gamepad, NaCl handles emulation/rendering/audio.

**Tech Stack:** C++ (snes9x 1.53), Samsung NaCl Pepper 63 SDK, PNaCl toolchain (pnacl-clang++), Docker (Ubuntu 20.04), PPAPI (Graphics3D/OpenGL ES 2.0, Audio, Input), JavaScript (menu UI, gamepad polling).

## Task 1: Setup Docker Build Environment

**Files:**
- Create: `nacl/Dockerfile`
- Create: `nacl/build.sh`

- [ ] **Step 1: Create nacl/Dockerfile**

```dockerfile
FROM ubuntu:20.04

ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y \
    wget unzip git build-essential python2 \
    && rm -rf /var/lib/apt/lists/*

# Create symlink for python (NaCl SDK expects 'python')
RUN ln -sf /usr/bin/python2 /usr/bin/python

# SDK will be mounted at /opt/nacl-sdk/pepper_63
# Source will be mounted at /src

WORKDIR /src
```

- [ ] **Step 2: Create nacl/build.sh**

```bash
#!/bin/bash
set -e

export NACL_SDK_ROOT=/opt/nacl-sdk/pepper_63
TOOLCHAIN=$NACL_SDK_ROOT/toolchain/linux_pnacl/bin

echo "=== Compiling snes4nacl for ARM ==="

cd /src/snes4nacl

# Clean previous build
rm -f *.pexe *.nexe

# Collect all source files
CORE_CPP=$(find . -maxdepth 1 -name "*.cpp" | sort)
APU_CPP=$(find apu -name "*.cpp" | sort)
NACL_CPP=$(find nacl -name "*.cpp" | sort)

ALL_CPP="$CORE_CPP $APU_CPP $NACL_CPP"

echo "Compiling $(echo $ALL_CPP | wc -w) source files..."

# Compile to PNaCl bitcode
$TOOLCHAIN/pnacl-clang++ \
    -I$NACL_SDK_ROOT/include \
    -I. -Iapu -Inacl \
    -DHAVE_STRINGS_H -DHAVE_STDINT_H -DRIGHTSHIFT_IS_SAR \
    -DVAR_CYCLES -DCPU_SHUTDOWN -DSPC700_SHUTDOWN \
    -DCORRECT_VRAM_READS -DUNZIP_SUPPORT \
    -DNOASM -DPIXEL_FORMAT=RGB565 \
    -DUSE_OPENGL -DNACL \
    -std=gnu++98 \
    -O2 \
    -o snes4nacl.pexe \
    $ALL_CPP \
    -L$NACL_SDK_ROOT/lib/pnacl/Release \
    -lppapi_gles2 -lppapi_cpp -lppapi -lnosys -lpthread

echo "=== Finalizing PNaCl ==="
$TOOLCHAIN/pnacl-finalize snes4nacl.pexe

echo "=== Translating to ARM ==="
$TOOLCHAIN/pnacl-translate -arch arm -o snes9x_arm.nexe snes4nacl.pexe

echo "=== Done ==="
ls -lh snes9x_arm.nexe

# Copy output
cp snes9x_arm.nexe /output/
cat > /output/app.nmf << 'EOF'
{
  "program": {
    "arm": {
      "url": "snes9x_arm.nexe"
    }
  }
}
EOF

echo "Output: /output/snes9x_arm.nexe + /output/app.nmf"
```

```cpp
void Input::setJoypadState(uint32_t bitmask) {
    // Map bitmask bits to snes9x joypad
    // Bit layout matches SNES joypad register:
    // 4=R, 5=L, 6=X, 7=A, 8=Right, 9=Left, 10=Down, 11=Up,
    // 12=Start, 13=Select, 14=Y, 15=B
    S9xReportButton(SNES_A_BUTTON,      (bitmask >> 7) & 1);
    S9xReportButton(SNES_B_BUTTON,      (bitmask >> 15) & 1);
    S9xReportButton(SNES_X_BUTTON,      (bitmask >> 6) & 1);
    S9xReportButton(SNES_Y_BUTTON,      (bitmask >> 14) & 1);
    S9xReportButton(SNES_L_BUTTON,      (bitmask >> 5) & 1);
    S9xReportButton(SNES_R_BUTTON,      (bitmask >> 4) & 1);
    S9xReportButton(SNES_START_BUTTON,  (bitmask >> 12) & 1);
    S9xReportButton(SNES_SELECT_BUTTON, (bitmask >> 13) & 1);
    S9xReportButton(SNES_UP_BUTTON,     (bitmask >> 11) & 1);
    S9xReportButton(SNES_DOWN_BUTTON,   (bitmask >> 10) & 1);
    S9xReportButton(SNES_LEFT_BUTTON,   (bitmask >> 9) & 1);
    S9xReportButton(SNES_RIGHT_BUTTON,  (bitmask >> 8) & 1);
}
```

- [ ] **Step 1: Rewrite js/emulator-snes.js as NaCl controller**

```javascript
var EmulatorSNES = (function() {
    'use strict';

    var naclModule = null;
    var isLoaded = false;
    var onReadyCallback = null;

    function init(container, onReady) {
        onReadyCallback = onReady;

        // Create embed element for NaCl
        var embed = document.createElement('embed');
        embed.id = 'nacl-snes';
        embed.setAttribute('type', 'application/x-nacl');
        embed.setAttribute('src', 'nacl/app.nmf');
        embed.style.width = '100%';
        embed.style.height = '100%';
        container.innerHTML = '';
        container.appendChild(embed);
        naclModule = embed;

        naclModule.addEventListener('load', function() {
            isLoaded = true;
            // Initialize snes9x inside NaCl
            naclModule.postMessage('init');
            if (onReadyCallback) onReadyCallback();
        });

        naclModule.addEventListener('message', function(e) {
            if (e.data === 'rom_loaded') {
                naclModule.postMessage('play');
                startFrameLoop();
            }
        });

        naclModule.addEventListener('error', function() {
            UI.toast('NaCl module failed to load');
        });

        naclModule.addEventListener('crash', function() {
            UI.toast('Emulator crashed');
        });
    }

    function loadROM(arrayBuffer) {
        if (!isLoaded) return;
        // Create blob URL from ArrayBuffer
        var blob = new Blob([arrayBuffer]);
        var url = URL.createObjectURL(blob);
        naclModule.postMessage('downloadThenLoadRom url:' + url);
    }

    var frameLoopId = null;
    function startFrameLoop() {
        function tick() {
            naclModule.postMessage('frame');
            frameLoopId = requestAnimationFrame(tick);
        }
        tick();
    }

    function stopFrameLoop() {
        if (frameLoopId) {
            cancelAnimationFrame(frameLoopId);
            frameLoopId = null;
        }
    }

    function setInput(bitmask) {
        if (!isLoaded || !naclModule) return;
        naclModule.postMessage('joypad b:' + bitmask);
    }

    function pause() {
        stopFrameLoop();
        naclModule.postMessage('pause');
    }

    function resume() {
        naclModule.postMessage('play');
        startFrameLoop();
    }

    function destroy() {
        stopFrameLoop();
        isLoaded = false;
        naclModule = null;
    }

    return {
        init: init,
        loadROM: loadROM,
        setInput: setInput,
        pause: pause,
        resume: resume,
        destroy: destroy
    };
})();
```
